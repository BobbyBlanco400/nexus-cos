// Attach to logo GameObject
public class LogoController : MonoBehaviour 
{
    private Pulse pulse;
    
    void Start() 
    {
        pulse = GetComponent<Pulse>();
    }
    
    void OnInteraction() 
    {
        pulse.TriggerSpike(intensity: 3f, duration: 0.5f);
    }
}
